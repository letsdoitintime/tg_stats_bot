# TG Stats Bot Requirements
# Pinned versions for production stability
# Reconciled against the running venv: 2026-07-22
#
# These pins describe what tgstats-bot / tgstats-migrations ACTUALLY run.
# Before this reconciliation the pins were stale by up to a major version, so
# `pip install -r requirements.txt` (Dockerfile, install_requirements.sh) would
# silently DOWNGRADE prod: redis 6.4->5.2, structlog 25.4->24.4, alembic
# 1.16->1.13, psycopg 3.2->3.1, pydantic 2.11->2.10, fastapi 0.116->0.115,
# uvicorn 0.35->0.32, celery 5.5->5.4, pydantic-settings 2.10->2.7.
# Keep this file in step with the venv; a bump belongs here and in the venv together.

# Core dependencies
python-telegram-bot[all]==22.5.*
sqlalchemy==2.0.*  # 2.0.51 verified: type round-trip unchanged (scripts/dbdriver_parity.py)
alembic==1.16.*
# 3.3.x verified to round-trip every type the schema uses — Decimal, date,
# naive/aware timestamptz, jsonb, bool, unicode — in BOTH directions, via
# scripts/dbdriver_parity.py. Bump asyncpg with it; they are separate codecs.
psycopg[binary]==3.3.*
# 0.31.x verified on the same harness through the ASYNC path, which is what
# the bot writes with. Proving psycopg says nothing about asyncpg.
asyncpg==0.31.*
pydantic==2.11.*
pydantic-settings==2.10.*
fastapi[all]==0.116.*
# starlette is a fastapi transitive. Under the fastapi pin above this line is
# redundant TODAY — fastapi 0.116.1 declares `starlette<0.48.0,>=0.40.0`, so it
# could not drift past 0.47.x anyway. It is here because newer fastapi DROPPED
# that upper bound: 0.139.2 declares a bare `starlette>=0.46.0` and resolves to
# starlette 1.3.1, a major. So this pin does nothing until the day someone bumps
# fastapi — which is exactly the day it matters. Bump it WITH fastapi, never alone.
starlette==0.47.*
uvicorn==0.35.*
# 26.x is a MAJOR, and structlog is imported by nearly every module here.
# Killer question was whether the configured chain still RENDERS the same —
# no test reads log lines, so a reordered key or changed exception folding
# would ship silently. scripts/structlog_parity.py renders a fixed event set
# through 4 chains, including the app's own ColoredConsoleRenderer and
# add_app_context: 35 lines, byte-identical across 25.4.0 -> 26.1.0.
structlog==26.1.*
# 2.15.x is a DATA change, not a helper bump: utils/features.py:40 counts
# emoji per message with is_emoji() and STORES the count, so the package's
# Unicode data version is part of the analytics contract.
# scripts/emoji_parity.py over every valid codepoint: 1393 -> 1400.
# The 7 additions are Unicode 17 (U+1F6D8 U+1FA8A U+1FA8E U+1FAC8 U+1FACD
# U+1FAEA U+1FAEF) and NOTHING was removed — so the change is append-only
# and no historical count can decrease. New messages using those brand-new
# emoji will count higher than an identical message did last month.
emoji==2.15.*
# 1.2.x verified to parse the real .env, .env.example and 11 syntax edge cases
# (quoting, interpolation, export kw, multiline, inline comments) byte-identically
# to 1.1.x. Reached via pydantic-settings, which is what actually loads config.
python-dotenv==1.2.*

# Step 2 additions
python-dateutil==2.9.*
# 2026.3 verified against every zone this app uses: scripts/tzdata_parity.py
# samples UTC offset, abbreviation and DST flag hourly through both switch
# windows plus weekly across two years, for the zones in group_settings
# (UTC, Europe/Sofia) and 12 others chosen for DST/non-DST, southern
# hemisphere, half-hour offsets and recent real-world rule changes.
# 110 transitions, zero differences. A changed IANA rule would silently
# move message counts between heatmap hours and days.
tzdata==2026.3
httpx==0.28.*
celery==5.5.*
redis==6.4.*
jinja2==3.1.*
